"use client";

import { useState } from "react";
import axios from "axios";
import { useRouter } from "next/navigation";

const API_BASE_URL = "http://localhost:8080"; // API Gateway

export default function StudentLoginPage() {
  const router = useRouter();

  const [email, setEmail] = useState("");
  const [password, setPassword] = useState("");
  const [error, setError] = useState<string | null>(null);
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setError(null);
    setLoading(true);

    try {
      const response = await axios.post(`${API_BASE_URL}/auth/login`, {
        email,
        password,
      });

      const { token, userId, fullName, roles } = response.data;

      // ✅ Persist login session
      localStorage.setItem("token", token);
      localStorage.setItem(
        "student",
        JSON.stringify({
          userId,
          email,
          fullName,
          roles,
        })
      );
       // ✅ 2. Store token in COOKIE for middleware
    document.cookie = `token=${token}; path=/;`;

      // ✅ Redirect to student dashboard
      router.push("/student/dashboard");
    } catch (err: any) {
      setError(
        err.response?.data?.message ||
          "Invalid email or password"
      );
    } finally {
      setLoading(false);
    }
  };

  return (
    <main className="container py-4">
      <div className="card p-4 mx-auto" style={{ maxWidth: "420px" }}>
        <h4 className="mb-3" style={{ color: "#051C36" }}>
          Student Login
        </h4>

        {error && (
          <div className="alert alert-danger">{error}</div>
        )}

        <form onSubmit={handleSubmit}>
          <div className="mb-3">
            <label className="form-label">Email</label>
            <input
              type="email"
              className="form-control"
              placeholder="you@i27academy.com"
              value={email}
              onChange={(e) => setEmail(e.target.value)}
              required
            />
          </div>

          <div className="mb-3">
            <label className="form-label">Password</label>
            <input
              type="password"
              className="form-control"
              value={password}
              onChange={(e) => setPassword(e.target.value)}
              required
            />
          </div>

          <button
            type="submit"
            className="btn btn-primary w-100"
            disabled={loading}
          >
            {loading ? "Logging in..." : "Login"}
          </button>
        </form>
      </div>
    </main>
  );
}
